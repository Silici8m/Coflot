fleet_interfaces — Messages

MissionRequest.msg

# MissionRequest.msg

# DESCRIPTION: Request definition for submitting a new mission to the fleet manager.

std_msgs/Header header

# Unique identifier for the mission (e.g., "mission_123")
string mission_id

# Priority level : higher values indicate higher priority (e.g., 1 for NORMAL, 2 for PRIORITAIRE, 3 for URGENTE)
int32 priority

# Ordered list of target poses the robot must visit
geometry_msgs/Pose[] waypoints

MissionState.msg

# MissionState.msg

# DESCRIPTION: Live status of a specific mission managed by the system.

string mission_id

# Priority level: 0 = Normal, 1 = High, 2 = Urgent
int32 priority

# Sequence of waypoints to visit
geometry_msgs/Pose[] waypoints

# ID of the robot performing the mission (empty string "" if unassigned)
string assigned_robot_id

# Lifecycle status.
# Values: PENDING, ASSIGNED, APPROACHING, WAITING, DELIVERING, 
#         SUSPENDING, DISCHARGING, FINISHED, FAILED
string status 

# Current target waypoint index.
# -1: Not started / 0: Moving to 1st waypoint / Increments upon departure.
int32 goal_waypoint_idx

# Time when the mission was originally requested
builtin_interfaces/Time request_stamp

RobotState.msg

# RobotState.msg

# DESCRIPTION: Telemetry and status data for a single robot.

# Unique identifier of the robot (namespace)
string robot_id

# Current operational mode (Not used)
string mode

# Battery level percentage (0.0 to 100.0)
float32 battery_pct

# Current map-referenced pose
geometry_msgs/Pose pose

# Timestamp of the data
builtin_interfaces/Time stamp
